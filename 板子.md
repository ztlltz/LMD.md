# 求数组中第k大的数
```c++
int Partition(vector<int>&a, int l, int r, int x){
	int i = l, j = r;
	do{
		while(a[i] < x) i++;
		while(a[j] > x) j--;
		if(i <= j){
			swap(a[i], a[j]);
			i++;
			j--;
		}
	}while(i <= j);
	return j;
}

int Select(vector<int>&a, int l, int r, int k){
	if(r - l + 1 <= 75){
		sort(a.begin() + l, a.begin() + r + 1);
		return a[l + k - 1];
	}
	for(int i = 0; i < (r - l + 1 - 4) / 5; ++i){
		sort(a.begin() + l + i * 5, a.begin() + l + i * 5 + 5);
		swap(a[l + i], a[l + i * 5 + 2]);
	}
	int mid = Select(a, l, l + (r - l + 1 - 4) / 5 - 1, ((r - l + 1 - 4) / 5 + 1) / 2);
	int pos = Partition(a, l, r, mid);
	int sz = pos - l + 1;
	if(k <= sz)
		return Select(a, l, pos, k);
	else return Select(a, pos + 1, r, k - sz);
}
```

# targan求连通分量
```c++
const int maxn = 200000+100;
struct Tarjan
{
	int V;
	struct edge
	{
		int u, v;
		int opp(int k) { return k==u?v:u; }
	};
	vector<edge> e;          //边
	vector<int> G[maxn];     //u连接的边
	int DFN[maxn], low[maxn], sk[maxn];
	int belong[maxn];
	bool instack[maxn];
	int top, Bcnt, Dindex;
	void init()
	{
		e.clear();
		for(int i=0; i<=V; i++) G[i].clear();
	}
	void add_edge(int u, int v)
	{
		e.push_back((edge){u, v});
		G[u].push_back(e.size()-1);
	}
	void dfs(int u)
	{
		DFN[u]=low[u]=++Dindex;
		instack[u] = true;
		sk[++top] = u;   //入栈
		for(int i=0; i<G[u].size(); i++)
		{
			int v = G[u][i];
			v = e[v].opp(u);          //u连接的边v
			if(!DFN[v])
			{
				dfs(v);
				low[u] = min(low[u], low[v]);
			}
			else if(instack[v])
				low[u] = min(low[u], DFN[v]);
		}
		if(DFN[u] == low[u])
		{
			Bcnt++;
			int v;
			do
			{
				v = sk[top--];
				instack[v] = false;
				belong[v]=Bcnt;
			}while(v!=u);
		}
	}
	void targan()
	{
		top = Bcnt = Dindex = 0;
		memset(DFN, 0, sizeof(DFN));
		for(int i=1; i<=V; i++)
			if(!DFN[i]) dfs(i);
	}
}ta;
```

# 线段树 
```c++
struct node{
	ll ans;
	int l;
	int r;
	ll lazy;
}tree[400005];
void pushup(int pos){
	tree[pos].ans=tree[pos*2].ans+tree[pos*2+1].ans;
	return ;
}

void pushdown(int pos){
	tree[pos*2].lazy+=tree[pos].lazy;
	tree[pos*2+1].lazy+=tree[pos].lazy;
	tree[pos*2].ans+=tree[pos].lazy*(tree[pos*2].r-tree[pos*2].l+1);
	tree[pos*2+1].ans+=tree[pos].lazy*(tree[pos*2+1].r-tree[pos*2+1].l+1);
	tree[pos].lazy=0;
}
void build(int pos,int l,int r){
	tree[pos].r=r;
	tree[pos].l=l;
	if(l==r){
		tree[pos].ans=num[l];
		return;
	}
	int mid=(l+r)/2;
	build(pos*2,l,mid);
	build(pos*2+1,mid+1,r);
	pushup(pos);
	return;
}
void update(int pos,int l,int r,int v){
	if(tree[pos].r<=r&&tree[pos].l>=l){
		tree[pos].lazy+=v;
		tree[pos].ans+=(tree[pos].r-tree[pos].l+1)*v;
		return;
	}
	pushdown(pos);
	int mid=(tree[pos].r+tree[pos].l)/2;
	if(mid>=l)update(pos*2,l,r,v);
	if(mid<r)update(pos*2+1,l,r,v);
	pushup(pos);
	return;
}
ll query(int pos,int l,int r){
	if(tree[pos].r<=r&&tree[pos].l>=l){
		return tree[pos].ans;
	}
	pushdown(pos);
	int mid=(tree[pos].r+tree[pos].l)/2;
	ll ans=0;
	if(mid>=l)ans+=query(pos*2,l,r);
	if(mid<r)ans+=query(pos*2+1,l,r);
	return ans;
}
```

```c++
{
    int ans = 1; 
    while(n)
    {
        if(n & 1) 
            ans *= a;
```

```c++
  for (int i = 2; i <= n; ++i) {
    if (!not_prime[i]) {
      pri.push_back(i);
    }
    for (int pri_j : pri) {

      not_prime[i * pri_j] = true;
      if (i % pri_j == 0) {
```
